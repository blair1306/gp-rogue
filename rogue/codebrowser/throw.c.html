61void throw(void)
62{
63 short wch, d;
64 boolean first_miss = 1;
65 object *weapon;
66 short dir, row, col;
67 object *monster;
68
69 while (!is_direction(dir = rgetchar(), &d)) {
70 sound_bell();
71 if (first_miss) {
72 message("direction? ", 0);
73 first_miss = 0;
74 }
75 }
76 check_message();
77 if (dir == CANCEL) {
78 return;
79 }
80 if ((wch = pack_letter("throw what?", WEAPON)) == CANCEL) {
81 return;
82 }
83 check_message();
84
85 if (!(weapon = get_letter_object(wch))) {
86 message("no such item.", 0);
87 return;
88 }
89 if ((weapon->in_use_flags & BEING_USED) && weapon->is_cursed) {
90 message(curse_message, 0);
91 return;
92 }
93 row = rogue.row;
94 col = rogue.col;
95
96 if ((weapon->in_use_flags & BEING_WIELDED) && (weapon->quantity <= 1)) {
97 unwield(rogue.weapon);
98 } else if (weapon->in_use_flags & BEING_WORN) {
99 mv_aquatars();
100 unwear(rogue.armor);
101 print_stats(STAT_ARMOR);
102 } else if (weapon->in_use_flags & ON_EITHER_HAND) {
103 un_put_on(weapon);
104 }
105 monster = get_thrown_at_monster(weapon, d, &row, &col);
106 mvaddch(rogue.row, rogue.col, rogue.fchar);
107 refresh();
108
109 if (rogue_can_see(row, col) && ((row != rogue.row) || (col != rogue.col))) {
110 mvaddch(row, col, get_dungeon_char(row, col));
111 }
112 if (monster) {
113 wake_up(monster);
114 check_gold_seeker(monster);
115
116 if (!throw_at_monster(monster, weapon)) {
117 flop_weapon(weapon, row, col);
118 }
119 } else {
120 flop_weapon(weapon, row, col);
121 }
122 vanish(weapon, 1, &rogue.pack);
123}
124
125throw_at_monster(monster, weapon)
126object *monster, *weapon;
127{
128 short damage, hit_chance;
129 short t;
130
131 hit_chance = get_hit_chance(weapon);
132 damage = get_weapon_damage(weapon);
133 if ((weapon->which_kind == ARROW) &&
134 (rogue.weapon && (rogue.weapon->which_kind == BOW))) {
135 damage += get_weapon_damage(rogue.weapon);
136 damage = ((damage * 2) / 3);
137 hit_chance += (hit_chance / 3);
138 } else if ((weapon->in_use_flags & BEING_WIELDED) &&
139 ((weapon->which_kind == DAGGER) ||
140 (weapon->which_kind == SHURIKEN) ||
141 (weapon->which_kind == DART))) {
142 damage = ((damage * 3) / 2);
143 hit_chance += (hit_chance / 3);
144 }
145 t = weapon->quantity;
146 weapon->quantity = 1;
147 sprintf(hit_message, "the %s", name_of(weapon));
148 weapon->quantity = t;
149
150 if (!rand_percent(hit_chance)) {
151 (void) strcat(hit_message, "misses ");
152 return (0);
153 }
154 s_con_mon(monster);
155 (void) strcat(hit_message, "hit ");
156 (void) mon_damage(monster, damage);
157 return (1);
158}

160object *
161get_thrown_at_monster(obj, dir, row, col)
162object *obj;
163short dir;
164short *row, *col;
165{
166 short orow, ocol;
167 short i, ch;
168
169 orow = *row;
170 ocol = *col;
171
172 ch = get_mask_char(obj->what_is);
173
174 for (i = 0; i < 24; i++) {
175 get_dir_rc(dir, row, col, 0);
176 if ((((*col <= 0) || (*col >= DCOLS - 1)) ||
177 (dungeon[*row][*col] == NOTHING)) ||
178 ((dungeon[*row][*col] & (HORWALL | VERTWALL | HIDDEN)) &&
179 (!(dungeon[*row][*col] & TRAP)))) {
180 *row = orow;
181 *col = ocol;
182 return (0);
183 }
184 if ((i != 0) && rogue_can_see(orow, ocol)) {
185 mvaddch(orow, ocol, get_dungeon_char(orow, ocol));
186 }
187 if (rogue_can_see(*row, *col)) {
188 if (!(dungeon[*row][*col] & MONSTER)) {
189 mvaddch(*row, *col, ch);
190 }
191 refresh();
192 }
193 orow = *row;
194 ocol = *col;
195 if (dungeon[*row][*col] & MONSTER) {
196 if (!imitating(*row, *col)) {
197 return (object_at(&level_monsters, *row, *col));
198 }
199 }
200 if (dungeon[*row][*col] & TUNNEL) {
201 i += 2;
202 }
203 }
204 return (0);
205}

207flop_weapon(weapon, row, col)
208object *weapon;
209short row, col;
210{
211 object *new_weapon, *monster;
212 short i = 0;
213 char msg[80];
214 boolean found = 0;
215 short mch, dch;
216 unsigned short mon;
217
218 while ((i < 9) && dungeon[row][col] & ~(FLOOR | TUNNEL | DOOR | MONSTER)) {
219 rand_around(i++, &row, &col);
220 if ((row > (DROWS - 2)) || (row < MIN_ROW) ||
221 (col > (DCOLS - 1)) || (col < 0) || (!dungeon[row][col]) ||
222 (dungeon[row][col] & ~(FLOOR | TUNNEL | DOOR | MONSTER))) {
223 continue;
224 }
225 found = 1;
226 break;
227 }
228
229 if (found || (i == 0)) {
230 new_weapon = alloc_object();
231 *new_weapon = *weapon;
232 new_weapon->in_use_flags = NOT_USED;
233 new_weapon->quantity = 1;
234 new_weapon->ichar = 'L';
235 place_at(new_weapon, row, col);
236 if (rogue_can_see(row, col) &&
237 ((row != rogue.row) || (col != rogue.col))) {
238 mon = dungeon[row][col] & MONSTER;
239 dungeon[row][col] &= (~MONSTER);
240 dch = get_dungeon_char(row, col);
241 if (mon) {
242 mch = mvinch(row, col);
243 if (monster = object_at(&level_monsters, row, col)) {
244 monster->trail_char = dch;
245 }
246 if ((mch < 'A') || (mch > 'Z')) {
247 mvaddch(row, col, dch);
248 }
249 } else {
250 mvaddch(row, col, dch);
251 }
252 dungeon[row][col] |= mon;
253 }
254 } else {
255 short t;
256
257 t = weapon->quantity;
258 weapon->quantity = 1;
259 sprintf(msg, "the %svanishes as it hits the ground",
260 name_of(weapon));
261 weapon->quantity = t;
262 message(msg, 0);
263 }
264}

266int rand_around(short i, short *r, short *c)
267{
268 static char pos[] = "\010\007\001\003\004\005\002\006\0";
269 static short row, col;
270 short j;
271
272 if (i == 0) {
273 short x, y, o, t;
274
275 row = *r;
276 col = *c;
277
278 o = get_rand(1, 8);
279
280 for (j = 0; j < 5; j++) {
281 x = get_rand(0, 8);
282 y = (x + o) % 9;
283 t = pos[x];
284 pos[x] = pos[y];
285 pos[y] = t;
286 }
287 }
288 switch ((short) pos[i]) {
289 case 0:
290 *r = row + 1;
291 *c = col + 1;
292 break;
293 case 1:
294 *r = row + 1;
295 *c = col - 1;
296 break;
297 case 2:
298 *r = row - 1;
299 *c = col + 1;
300 break;
301 case 3:
302 *r = row - 1;
303 *c = col - 1;
304 break;
305 case 4:
306 *r = row;
307 *c = col + 1;
308 break;
309 case 5:
310 *r = row + 1;
311 *c = col;
312 break;
313 case 6:
314 *r = row;
315 *c = col;
316 break;
317 case 7:
318 *r = row - 1;
319 *c = col;
320 break;
321 case 8:
322 *r = row;
323 *c = col - 1;
324 break;
325 }
