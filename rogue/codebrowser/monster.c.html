134put_mons()
135{
136 short i;
137 short n;
138 object *monster;
139 short row, col;
140
141 n = get_rand(4, 6);
142
143 for (i = 0; i < n; i++) {
144 monster = gr_monster((object *) 0, 0);
145 if ((monster->m_flags & WANDERS) && coin_toss()) {
146 wake_up(monster);
147 }
148 gr_row_col(&row, &col, (FLOOR | TUNNEL | STAIRS | OBJECT));
149 put_m_at(row, col, monster);
150 }

153object *
154gr_monster(monster, mn)
155register object *monster;
156register mn;
157{
158 if (!monster) {
159 monster = alloc_object();
160
161 for (;;) {
162 mn = get_rand(0, MONSTERS - 1);
163 if ((cur_level >= mon_tab[mn].first_level) &&
164 (cur_level <= mon_tab[mn].last_level)) {
165 break;
166 }
167 }
168 }
169 *monster = mon_tab[mn];
170 if (monster->m_flags & IMITATES) {
171 monster->disguise = gr_obj_char();
172 }
173 if (cur_level > (AMULET_LEVEL + 2)) {
174 monster->m_flags |= HASTED;
175 }
176 monster->trow = NO_ROOM;
177 return (monster);
178}

180mv_mons()
181{
182 register object *monster, *next_monster;
183 boolean flew;
184
185 if (haste_self % 2) {
186 return;
187 }
188
189 monster = level_monsters.next_monster;
190
191 while (monster) {
192 next_monster = monster->next_monster;
193 mon_disappeared = 0;
194 if (monster->m_flags & HASTED) {
195 mv_1_monster(monster, rogue.row, rogue.col);
196 if (mon_disappeared) {
197 goto NM;
198 }
199 } else if (monster->m_flags & SLOWED) {
200 monster->slowed_toggle = !monster->slowed_toggle;
201 if (monster->slowed_toggle) {
202 goto NM;
203 }
204 }
205 if ((monster->m_flags & CONFUSED) && move_confused(monster)) {
206 goto NM;
207 }
208 flew = 0;
209 if ((monster->m_flags & FLIES) &&
210 !(monster->m_flags & NAPPING) &&
211 !mon_can_go(monster, rogue.row, rogue.col)) {
212 flew = 1;
213 mv_1_monster(monster, rogue.row, rogue.col);
214 if (mon_disappeared) {
215 goto NM;
216 }
217 }
218 if (!(flew && mon_can_go(monster, rogue.row, rogue.col))) {
219 mv_1_monster(monster, rogue.row, rogue.col);
220 }
221NM:
222 monster = next_monster;
223 }
224}

226party_monsters(rn, n)
227int rn, n;
228{
229 short i, j;
230 short row, col;
231 object *monster;
232 boolean found;
233
234 n += n;
235
236 for (i = 0; i < MONSTERS; i++) {
237 mon_tab[i].first_level -= (cur_level % 3);
238 }
239 for (i = 0; i < n; i++) {
240 if (no_room_for_monster(rn)) {
241 break;
242 }
243 for (j = found = 0; ((!found) && (j < 250)); j++) {
244 row = get_rand(rooms[rn].top_row + 1,
245 rooms[rn].bottom_row - 1);
246 col = get_rand(rooms[rn].left_col + 1,
247 rooms[rn].right_col - 1);
248 if ((!(dungeon[row][col] & MONSTER)) &&
249 (dungeon[row][col] & (FLOOR | TUNNEL))) {
250 found = 1;
251 }
252 }
253 if (found) {
254 monster = gr_monster((object *) 0, 0);
255 if (!(monster->m_flags & IMITATES)) {
256 monster->m_flags |= WAKENS;
257 }
258 put_m_at(row, col, monster);
259 }
260 }
261 for (i = 0; i < MONSTERS; i++) {
262 mon_tab[i].first_level += (cur_level % 3);
263 }
264}

267register row, col;
268{

272 if ((!(detect_monster || see_invisible || r_see_invisible) &&

274 return (monster->trail_char);

276 if (monster->m_flags & IMITATES) {
277 return (monster->disguise);

299mv_1_monster(monster, row, col)
300register object *monster;
301short row, col;
302{
303 short i, n;
304 boolean tried[6];
305
306 if (monster->m_flags & ASLEEP) {
307 if (monster->m_flags & NAPPING) {
308 if (--monster->nap_length <= 0) {
309 monster->m_flags &= (~(NAPPING | ASLEEP));
310 }
311 return;
312 }
313 if ((monster->m_flags & WAKENS) &&
314 rogue_is_around(monster->row, monster->col) &&
315 rand_percent(((stealthy > 0) ?
316 (WAKE_PERCENT / (STEALTH_FACTOR + stealthy)) :
317 WAKE_PERCENT))) {
318 wake_up(monster);
319 }
320 return;
321 } else if (monster->m_flags & ALREADY_MOVED) {
322 monster->m_flags &= (~ALREADY_MOVED);
323 return;
324 }
325 if ((monster->m_flags & FLITS) && flit(monster)) {
326 return;
327 }
328 if ((monster->m_flags & STATIONARY) &&
329 (!mon_can_go(monster, rogue.row, rogue.col))) {
330 return;
331 }
332 if (monster->m_flags & FREEZING_ROGUE) {
333 return;
334 }
335 if ((monster->m_flags & CONFUSES) && m_confuse(monster)) {
336 return;
337 }
338 if (mon_can_go(monster, rogue.row, rogue.col)) {
339 mon_hit(monster);
340 return;
341 }
342 if ((monster->m_flags & FLAMES) && flame_broil(monster)) {
343 return;
344 }
345 if ((monster->m_flags & SEEKS_GOLD) && seek_gold(monster)) {
346 return;
347 }
348 if ((monster->trow == monster->row) &&
349 (monster->tcol == monster->col)) {
350 monster->trow = NO_ROOM;
351 } else if (monster->trow != NO_ROOM) {
352 row = monster->trow;
353 col = monster->tcol;
354 }
355 if (monster->row > row) {
356 row = monster->row - 1;
357 } else if (monster->row < row) {
358 row = monster->row + 1;
359 }
360 if ((dungeon[row][monster->col] & DOOR) &&
361 mtry(monster, row, monster->col)) {
362 return;
363 }
364 if (monster->col > col) {
365 col = monster->col - 1;
366 } else if (monster->col < col) {
367 col = monster->col + 1;
368 }
369 if ((dungeon[monster->row][col] & DOOR) &&
370 mtry(monster, monster->row, col)) {
371 return;
372 }
373 if (mtry(monster, row, col)) {
374 return;
375 }
376
377 for (i = 0; i <= 5; i++) {
378 tried[i] = 0;
379 }
380
381 for (i = 0; i < 6; i++) {
382NEXT_TRY:
383 n = get_rand(0, 5);
384 switch (n) {
385 case 0:
386 if (!tried[n] && mtry(monster, row, monster->col - 1)) {
387 goto O;
388 }
389 break;
390 case 1:
391 if (!tried[n] && mtry(monster, row, monster->col)) {
392 goto O;
393 }
394 break;
395 case 2:
396 if (!tried[n] && mtry(monster, row, monster->col + 1)) {
397 goto O;
398 }
399 break;
400 case 3:
401 if (!tried[n] && mtry(monster, monster->row - 1, col)) {
402 goto O;
403 }
404 break;
405 case 4:
406 if (!tried[n] && mtry(monster, monster->row, col)) {
407 goto O;
408 }
409 break;
410 case 5:
411 if (!tried[n] && mtry(monster, monster->row + 1, col)) {
412 goto O;
413 }
414 break;
415 }
416 if (!tried[n]) {
417 tried[n] = 1;
418 } else {
419 goto NEXT_TRY;
420 }
421 }
422O:
423 if ((monster->row == monster->o_row) && (monster->col == monster->o_col)) {
424 if (++ (monster->o) > 4) {
425 if ((monster->trow == NO_ROOM) &&
426 (!mon_sees(monster, rogue.row, rogue.col))) {
427 monster->trow = get_rand(1, (DROWS - 2));
428 monster->tcol = get_rand(0, (DCOLS - 1));
429 } else {
430 monster->trow = NO_ROOM;
431 monster->o = 0;
432 }
433 }
434 } else {
435 monster->o_row = monster->row;
436 monster->o_col = monster->col;
437 monster->o = 0;
438 }
439}

441mtry(monster, row, col)
442register object *monster;
443register short row, col;
444{
445 if (mon_can_go(monster, row, col)) {
446 move_mon_to(monster, row, col);
447 return (1);
448 }
449 return (0);
450}
451
452move_mon_to(monster, row, col)
453register object *monster;
454register short row, col;
455{
456 short c;
457 register mrow, mcol;
458
459 mrow = monster->row;
460 mcol = monster->col;
461
462 dungeon[mrow][mcol] &= ~MONSTER;
463 dungeon[row][col] |= MONSTER;
464
465 c = mvinch(mrow, mcol);
466
467 if ((c >= 'A') && (c <= 'Z')) {
468 if (!detect_monster) {
469 mvaddch(mrow, mcol, monster->trail_char);
470 } else {
471 if (rogue_can_see(mrow, mcol)) {
472 mvaddch(mrow, mcol, monster->trail_char);
473 } else {
474 if (monster->trail_char == '.') {
475 monster->trail_char = ' ';
476 }
477 mvaddch(mrow, mcol, monster->trail_char);
478 }
479 }
480 }
481 monster->trail_char = mvinch(row, col);
482 if (!blind && (detect_monster || rogue_can_see(row, col))) {
483 if ((!(monster->m_flags & INVISIBLE) ||
484 (detect_monster || see_invisible || r_see_invisible))) {
485 mvaddch(row, col, gmc(monster));
486 }
487 }
488 if ((dungeon[row][col] & DOOR) &&
489 (get_room_number(row, col) != cur_room) &&
490 (dungeon[mrow][mcol] == FLOOR) && !blind) {
491 mvaddch(mrow, mcol, ' ');
492 }
493 if (dungeon[row][col] & DOOR) {
494 dr_course(monster, ((dungeon[mrow][mcol] & TUNNEL) ? 1 : 0),
495 row, col);
496 } else {
497 monster->row = row;
498 monster->col = col;
499 }
500}
501
502mon_can_go(monster, row, col)
503register object *monster;
504register short row, col;
505{
506 object *obj;
507 short dr, dc;
508
509 dr = monster->row - row; /* check if move distance > 1 */
510 if ((dr >= 2) || (dr <= -2)) {
511 return (0);
512 }
513 dc = monster->col - col;
514 if ((dc >= 2) || (dc <= -2)) {
515 return (0);
516 }
517 if ((!dungeon[monster->row][col]) || (!dungeon[row][monster->col])) {
518 return (0);
519 }
520 if ((!is_passable(row, col)) || (dungeon[row][col] & MONSTER)) {
521 return (0);
522 }
523 if ((monster->row != row) && (monster->col != col) && ((dungeon[row][col]&DOOR) ||
524 (dungeon[monster->row][monster->col]&DOOR))) {
525 return (0);
526 }
527 if (!(monster->m_flags & (FLITS | CONFUSED | CAN_FLIT)) &&
528 (monster->trow == NO_ROOM)) {
529 if ((monster->row < rogue.row) && (row < monster->row)) {
530 return (0);
531 }
532 if ((monster->row > rogue.row) && (row > monster->row)) {
533 return (0);
534 }
535 if ((monster->col < rogue.col) && (col < monster->col)) {
536 return (0);
537 }
538 if ((monster->col > rogue.col) && (col > monster->col)) {
539 return (0);
540 }
541 }
542 if (dungeon[row][col] & OBJECT) {
543 obj = object_at(&level_objects, row, col);
544 if ((obj->what_is == SCROL) && (obj->which_kind == SCARE_MONSTER)) {
545 return (0);
546 }
547 }
548 return (1);
549}

559wake_room(rn, entering, row, col)
560short rn;
561boolean entering;
562short row, col;
563{
564 object *monster;
565 short wake_percent;
566 boolean in_room;
567
568 wake_percent = (rn == party_room) ? PARTY_WAKE_PERCENT : WAKE_PERCENT;
569 if (stealthy > 0) {
570 wake_percent /= (STEALTH_FACTOR + stealthy);
571 }
572
573 monster = level_monsters.next_monster;
574
575 while (monster) {
576 in_room = (rn == get_room_number(monster->row, monster->col));
577 if (in_room) {
578 if (entering) {
579 monster->trow = NO_ROOM;
580 } else {
581 monster->trow = row;
582 monster->tcol = col;
583 }
584 }
585 if ((monster->m_flags & WAKENS) &&
586 (rn == get_room_number(monster->row, monster->col))) {
587 if (rand_percent(wake_percent)) {
588 wake_up(monster);
589 }
590 }
591 monster = monster->next_monster;
592 }

613rogue_is_around(row, col)
614register row, col;
615{
616 short rdif, cdif, retval;
617
618 rdif = row - rogue.row;
619 cdif = col - rogue.col;
620
621 retval = (rdif >= -1) && (rdif <= 1) && (cdif >= -1) && (cdif <= 1);
622 return (retval);
623}

625wanderer()
626{
627 object *monster;
628 short row, col, i;
629 boolean found = 0;
630
631 for (i = 0; ((i < 15) && (!found)); i++) {
632 monster = gr_monster((object *) 0, 0);
633 if (!(monster->m_flags & (WAKENS | WANDERS))) {
634 free_object(monster);
635 } else {
636 found = 1;
637 }
638 }
639 if (found) {
640 found = 0;
641 wake_up(monster);
642 for (i = 0; ((i < 25) && (!found)); i++) {
643 gr_row_col(&row, &col, (FLOOR | TUNNEL | STAIRS | OBJECT));
644 if (!rogue_can_see(row, col)) {
645 put_m_at(row, col, monster);
646 found = 1;
647 }
648 }
649 if (!found) {
650 free_object(monster);
651 }
652 }
653}

655show_monsters()
656{
657 object *monster;
658
659 detect_monster = 1;
660
661 if (blind) {
662 return;
663 }
664 monster = level_monsters.next_monster;
665
666 while (monster) {
667 mvaddch(monster->row, monster->col, monster->m_char);
668 if (monster->m_flags & IMITATES) {
669 monster->m_flags &= (~IMITATES);
670 monster->m_flags |= WAKENS;
671 }
672 monster = monster->next_monster;
673 }

675
676create_monster()
677{
678 short row, col;
679 short i;
680 boolean found = 0;
681 object *monster;
682
683 row = rogue.row;
684 col = rogue.col;
685
686 for (i = 0; i < 9; i++) {
687 rand_around(i, &row, &col);
688 if (((row == rogue.row) && (col = rogue.col)) ||
689 (row < MIN_ROW) || (row > (DROWS - 2)) ||
690 (col < 0) || (col > (DCOLS - 1))) {
691 continue;
692 }
693 if ((!(dungeon[row][col] & MONSTER)) &&
694 (dungeon[row][col] & (FLOOR | TUNNEL | STAIRS | DOOR))) {
695 found = 1;
696 break;
697 }
698 }
699 if (found) {
700 monster = gr_monster((object *) 0, 0);
701 put_m_at(row, col, monster);
702 mvaddch(row, col, gmc(monster));
703 if (monster->m_flags & (WANDERS | WAKENS)) {
704 wake_up(monster);
705 }
706 } else {
707 message("you hear a faint cry of anguish in the distance", 0);

864 return (1);
