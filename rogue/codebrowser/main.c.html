59#include "rogue.h"
60
61extern short party_room;
62
63main(argc, argv)
64int argc;
65char *argv[];
66{
67 if (init(argc, argv)) { /* restored game */
68 goto PL;
69 }
70
71 for (;;) {
72 clear_level();
73 make_level();
74 put_objects();
75 put_stairs();
76 add_traps();
77 put_mons();
78 put_player(party_room);
79 print_stats(STAT_ALL);
80PL:
81 play_level();
82 free_stuff(&level_objects);
83 free_stuff(&level_monsters);
84 }
85}
