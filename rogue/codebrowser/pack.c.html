60object *
61add_to_pack(obj, pack, condense)
62object *obj, *pack;
63{
64 object *op;
65
66 if (condense) {
67 if (op = check_duplicate(obj, pack)) {
68 free_object(obj);
69 return (op);
70 } else {
71 obj->ichar = next_avail_ichar();
72 }
73 }
74 if (pack->next_object == 0) {
75 pack->next_object = obj;
76 } else {
77 op = pack->next_object;
78
79 while (op->next_object) {
80 op = op->next_object;
81 }
82 op->next_object = obj;
83 }
84 obj->next_object = 0;
85 return (obj);
86}
87
88take_from_pack(obj, pack)
89object *obj, *pack;
90{
91 while (pack->next_object != obj) {
92 pack = pack->next_object;
93 }
94 pack->next_object = pack->next_object->next_object;
95}

97/* Note: *status is set to 0 if the rogue attempts to pick up a scroll
98 * of scare-monster and it turns to dust. *status is otherwise set to 1.
99 */
100
101object *
102pick_up(row, col, status)
103short *status;
104{
105 object *obj;
106
107 *status = 1;
108
109 if (levitate) {
110 message("you're floating in the air!", 0);
111 return ((object *) 0);
112 }
113 obj = object_at(&level_objects, row, col);
114 if (!obj) {
115 message("pick_up(): inconsistent", 1);
116 return (obj);
117 }
118 if ((obj->what_is == SCROL) &&
119 (obj->which_kind == SCARE_MONSTER) &&
120 obj->picked_up) {
121 message("the scroll turns to dust as you pick it up", 0);
122 dungeon[row][col] &= (~OBJECT);
123 vanish(obj, 0, &level_objects);
124 *status = 0;
125 if (id_scrolls[SCARE_MONSTER].id_status == UNIDENTIFIED) {
126 id_scrolls[SCARE_MONSTER].id_status = IDENTIFIED;
127 }
128 return ((object *) 0);
129 }
130 if (obj->what_is == GOLD) {
131 rogue.gold += obj->quantity;
132 dungeon[row][col] &= ~(OBJECT);
133 take_from_pack(obj, &level_objects);
134 print_stats(STAT_GOLD);
135 return (obj); /* obj will be free_object()ed in caller */
136 }
137 if (pack_count(obj) >= MAX_PACK_COUNT) {
138 message("pack too full", 1);
139 return ((object *) 0);
140 }
141 dungeon[row][col] &= ~(OBJECT);
142 take_from_pack(obj, &level_objects);
143 obj = add_to_pack(obj, &rogue.pack, 1);
144 obj->picked_up = 1;
145 return (obj);
146}

148drop()
149{
150 object *obj, *new;
151 short ch;
152 char desc[DCOLS];
153
154 if (dungeon[rogue.row][rogue.col] & (OBJECT | STAIRS | TRAP)) {
155 message("there's already something there", 0);
156 return;
157 }
158 if (!rogue.pack.next_object) {
159 message("you have nothing to drop", 0);
160 return;
161 }
162 if ((ch = pack_letter("drop what?", ALL_OBJECTS)) == CANCEL) {
163 return;
164 }
165 if (!(obj = get_letter_object(ch))) {
166 message("no such item.", 0);
167 return;
168 }
169 if (obj->in_use_flags & BEING_WIELDED) {
170 if (obj->is_cursed) {
171 message(curse_message, 0);
172 return;
173 }
174 unwield(rogue.weapon);
175 } else if (obj->in_use_flags & BEING_WORN) {
176 if (obj->is_cursed) {
177 message(curse_message, 0);
178 return;
179 }
180 mv_aquatars();
181 unwear(rogue.armor);
182 print_stats(STAT_ARMOR);
183 } else if (obj->in_use_flags & ON_EITHER_HAND) {
184 if (obj->is_cursed) {
185 message(curse_message, 0);
186 return;
187 }
188 un_put_on(obj);
189 }
190 obj->row = rogue.row;
191 obj->col = rogue.col;
192
193 if ((obj->quantity > 1) && (obj->what_is != WEAPON)) {
194 obj->quantity--;
195 new = alloc_object();
196 *new = *obj;
197 new->quantity = 1;
198 obj = new;
199 } else {
200 obj->ichar = 'L';
201 take_from_pack(obj, &rogue.pack);
202 }
203 place_at(obj, rogue.row, rogue.col);
204 (void) strcpy(desc, "dropped ");
205 get_desc(obj, desc + 8);
206 message(desc, 0);
207 (void) reg_move();
208}

210object *
211check_duplicate(obj, pack)
212object *obj, *pack;
213{
214 object *op;
215
216 if (!(obj->what_is & (WEAPON | FOOD | SCROL | POTION))) {
217 return (0);
218 }
219 if ((obj->what_is == FOOD) && (obj->which_kind == FRUIT)) {
220 return (0);
221 }
222 op = pack->next_object;
223
224 while (op) {
225 if ((op->what_is == obj->what_is) &&
226 (op->which_kind == obj->which_kind)) {
227
228 if ((obj->what_is != WEAPON) ||
229 ((obj->what_is == WEAPON) &&
230 ((obj->which_kind == ARROW) ||
231 (obj->which_kind == DAGGER) ||
232 (obj->which_kind == DART) ||
233 (obj->which_kind == SHURIKEN)) &&
234 (obj->quiver == op->quiver))) {
235 op->quantity += obj->quantity;
236 return (op);
237 }
238 }
239 op = op->next_object;
240 }
241 return (0);
242}

244next_avail_ichar()
245{
246 register object *obj;
247 register i;
248 boolean ichars[26];
249
250 for (i = 0; i < 26; i++) {
251 ichars[i] = 0;
252 }
253 obj = rogue.pack.next_object;
254 while (obj) {
255 ichars[(obj->ichar - 'a')] = 1;
256 obj = obj->next_object;
257 }
258 for (i = 0; i < 26; i++) {
259 if (!ichars[i]) {
260 return (i + 'a');
261 }
262 }
263 return ('?');
264}

271pack_letter(prompt, mask)
272char *prompt;
273unsigned short mask;
274{
275 short ch;
276 unsigned short tmask = mask;
277
278 if (!mask_pack(&rogue.pack, mask)) {
279 message("nothing appropriate", 0);
280 return (CANCEL);
281 }
282 for (;;) {
283
284 message(prompt, 0);
285
286 for (;;) {
287 ch = rgetchar();
288 if (!is_pack_letter(&ch, &mask)) {
289 sound_bell();
290 } else {
291 break;
292 }
293 }
294
295 if (ch == LIST) {
296 check_message();
297 mask = tmask;
298 inventory(&rogue.pack, mask);
299 } else {
300 break;
301 }
302 mask = tmask;
303 }
304 check_message();
305 return (ch);
306}

308take_off()
309{
310 char desc[DCOLS];
311 object *obj;
312
313 if (rogue.armor) {
314 if (rogue.armor->is_cursed) {
315 message(curse_message, 0);
316 } else {
317 mv_aquatars();
318 obj = rogue.armor;
319 unwear(rogue.armor);
320 (void) strcpy(desc, "was wearing ");
321 get_desc(obj, desc + 12);
322 message(desc, 0);
323 print_stats(STAT_ARMOR);
324 (void) reg_move();
325 }
326 } else {
327 message("not wearing any", 0);
328 }

331wear()
332{
333 short ch;
334 register object *obj;
335 char desc[DCOLS];
336
337 if (rogue.armor) {
338 message("your already wearing some", 0);
339 return;
340 }
341 ch = pack_letter("wear what?", ARMOR);
342
343 if (ch == CANCEL) {
344 return;
345 }
346 if (!(obj = get_letter_object(ch))) {
347 message("no such item.", 0);
348 return;
349 }
350 if (obj->what_is != ARMOR) {
351 message("you can't wear that", 0);
352 return;
353 }
354 obj->identified = 1;
355 (void) strcpy(desc, "wearing ");
356 get_desc(obj, desc + 8);
357 message(desc, 0);
358 do_wear(obj);
359 print_stats(STAT_ARMOR);
360 (void) reg_move();
361}

380wield()
381{
382 short ch;
383 register object *obj;
384 char desc[DCOLS];
385
386 if (rogue.weapon && rogue.weapon->is_cursed) {
387 message(curse_message, 0);
388 return;
389 }
390 ch = pack_letter("wield what?", WEAPON);
391
392 if (ch == CANCEL) {
393 return;
394 }
395 if (!(obj = get_letter_object(ch))) {
396 message("No such item.", 0);
397 return;
398 }
399 if (obj->what_is & (ARMOR | RING)) {
400 sprintf(desc, "you can't wield %s",
401 ((obj->what_is == ARMOR) ? "armor" : "rings"));
402 message(desc, 0);
403 return;
404 }
405 if (obj->in_use_flags & BEING_WIELDED) {
406 message("in use", 0);
407 } else {
408 unwield(rogue.weapon);
409 (void) strcpy(desc, "wielding ");
410 get_desc(obj, desc + 9);
411 message(desc, 0);
412 do_wield(obj);
413 (void) reg_move();
414 }
415}

433call_it()
434{
435 short ch;
436 register object *obj;
437 struct id *id_table;
438 char buf[MAX_TITLE_LENGTH + 2];
439
440 ch = pack_letter("call what?", (SCROL | POTION | WAND | RING));
441
442 if (ch == CANCEL) {
443 return;
444 }
445 if (!(obj = get_letter_object(ch))) {
446 message("no such item.", 0);
447 return;
448 }
449 if (!(obj->what_is & (SCROL | POTION | WAND | RING))) {
450 message("surely you already know what that's called", 0);
451 return;
452 }
453 id_table = get_id_table(obj);
454
455 if (get_input_line("call it:", "", buf, id_table[obj->which_kind].title, 1, 1)) {
456 id_table[obj->which_kind].id_status = CALLED;
457 (void) strcpy(id_table[obj->which_kind].title, buf);
458 }
459}

461pack_count(new_obj)
462object *new_obj;
463{
464 object *obj;
465 short count = 0;
466
467 obj = rogue.pack.next_object;
468
469 while (obj) {
470 if (obj->what_is != WEAPON) {
471 count += obj->quantity;
472 } else if (!new_obj) {
473 count++;
474 } else if ((new_obj->what_is != WEAPON) ||
475 ((obj->which_kind != ARROW) &&
476 (obj->which_kind != DAGGER) &&
477 (obj->which_kind != DART) &&
478 (obj->which_kind != SHURIKEN)) ||
479 (new_obj->which_kind != obj->which_kind) ||
480 (obj->quiver != new_obj->quiver)) {
481 count++;
482 }
483 obj = obj->next_object;
484 }
485 return (count);
486}

488boolean
489mask_pack(pack, mask)
490object *pack;
491unsigned short mask;
492{
493 while (pack->next_object) {
494 pack = pack->next_object;
495 if (pack->what_is & mask) {
496 return (1);
497 }
498 }
499 return (0);
500}
501
502is_pack_letter(c, mask)
503short *c;
504unsigned short *mask;
505{
506 if (((*c == '?') || (*c == '!') || (*c == ':') || (*c == '=') ||
507 (*c == ')') || (*c == ']') || (*c == '/') || (*c == ','))) {
508 switch (*c) {
509 case '?':
510 *mask = SCROL;
511 break;
512 case '!':
513 *mask = POTION;
514 break;
515 case ':':
516 *mask = FOOD;
517 break;
518 case ')':
519 *mask = WEAPON;
520 break;
521 case ']':
522 *mask = ARMOR;
523 break;
524 case '/':
525 *mask = WAND;
526 break;
527 case '=':
528 *mask = RING;
529 break;
530 case ',':
531 *mask = AMULET;
532 break;
533 }
534 *c = LIST;
535 return (1);
536 }
537 return (((*c >= 'a') && (*c <= 'z')) || (*c == CANCEL) || (*c == LIST));
538}

540has_amulet()
541{
542 return (mask_pack(&rogue.pack, AMULET));

544
545kick_into_pack()
546{
547 object *obj;
548 char desc[DCOLS];
549 short n, stat;
550
551 if (!(dungeon[rogue.row][rogue.col] & OBJECT)) {
552 message("nothing here", 0);
553 } else {
554 if (obj = pick_up(rogue.row, rogue.col, &stat)) {
555 get_desc(obj, desc);
556 if (obj->what_is == GOLD) {
557 message(desc, 0);
558 free_object(obj);
559 } else {
560 n = strlen(desc);
561 desc[n] = '(';
562 desc[n + 1] = obj->ichar;
563 desc[n + 2] = ')';
564 desc[n + 3] = 0;
565 message(desc, 0);
566 }
567 }
568 if (obj || (!stat)) {
569 (void) reg_move();
570 }
571 }
