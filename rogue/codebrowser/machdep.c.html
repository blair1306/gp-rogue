53/* Included in this file are all system dependent routines. Extensive use
54 * of #ifdef's will be used to compile the appropriate code on each system:
55 *
56 * UNIX: all UNIX systems.
57 * UNIX_BSD4_2: UNIX BSD 4.2 and later, UTEK, (4.1 BSD too?)
58 * UNIX_SYSV: UNIX system V
59 * UNIX_V7: UNIX version 7
60 *
61 * All UNIX code should be included between the single "#ifdef UNIX" at the
62 * top of this file, and the "#endif" at the bottom.
63 *
64 * To change a routine to include a new UNIX system, simply #ifdef the
65 * existing routine, as in the following example:
66 *
67 * To make a routine compatible with UNIX system 5, change the first
68 * function to the second:
69 *
70 * md_function()
71 * {
72 * code;
73 * }
74 *
75 * md_function()
76 * {
77 * #ifdef UNIX_SYSV
78 * sys5code;
79 * #else
80 * code;
81 * #endif
82 * }
83 *
84 * Appropriate variations of this are of course acceptible.
85 * The use of "#elseif" is discouraged because of non-portability.
86 * If the correct #define doesn't exist, "UNIX_SYSV" in this case, make it up
87 * and insert it in the list at the top of the file. Alter the CFLAGS
88 * in you Makefile appropriately.
89 *
90 */

140/* md_control_keyboard():
141 *
142 * This routine is much like md_cbreak_no_echo_nonl() below. It sets up the
143 * keyboard for appropriate input. Specifically, it prevents the tty driver
144 * from stealing characters. For example, ^Y is needed as a command
145 * character, but the tty driver intercepts it for another purpose. Any
146 * such behavior should be stopped. This routine could be avoided if
147 * we used RAW mode instead of CBREAK. But RAW mode does not allow the
148 * generation of keyboard signals, which the program uses.
149 *
150 * The parameter 'mode' when true, indicates that the keyboard should
151 * be set up to play rogue. When false, it should be restored if
152 * necessary.
153 *
154 * This routine is not strictly necessary and may be stubbed. This may
155 * cause certain command characters to be unavailable.
156 */
157
158md_control_keybord(mode)
159boolean mode;
160{
161 static boolean called_before = 0;
162#ifdef UNIX_BSD4_2
163 static struct ltchars ltc_orig;
164 static struct tchars tc_orig;
165 struct ltchars ltc_temp;
166 struct tchars tc_temp;
167#endif
168#ifdef UNIX_SYSV
169 static struct termio _oldtty;
170 struct termio _tty;
171#endif
172
173 if (!called_before) {
174 called_before = 1;
175#ifdef UNIX_BSD4_2
176 ioctl(0, TIOCGETC, &tc_orig);
177 ioctl(0, TIOCGLTC, &ltc_orig);
178#endif
179#ifdef UNIX_SYSV
180 ioctl(0, TCGETA, &_oldtty);
181#endif
182 }
183#ifdef UNIX_BSD4_2
184 ltc_temp = ltc_orig;
185 tc_temp = tc_orig;
186#endif
187#ifdef UNIX_SYSV
188 _tty = _oldtty;
189#endif
190
191 if (!mode) {
192#ifdef UNIX_BSD4_2
193 ltc_temp.t_suspc = ltc_temp.t_dsuspc = -1;
194 ltc_temp.t_rprntc = ltc_temp.t_flushc = -1;
195 ltc_temp.t_werasc = ltc_temp.t_lnextc = -1;
196 tc_temp.t_startc = tc_temp.t_stopc = -1;
197#endif
198#ifdef UNIX_SYSV
199 _tty.c_cc[VSWTCH] = CNSWTCH;
200#endif
201 }
202#ifdef UNIX_BSD4_2
203 ioctl(0, TIOCSETC, &tc_temp);
204 ioctl(0, TIOCSLTC, &ltc_temp);
205#endif
206#ifdef UNIX_SYSV
207 ioctl(0, TCSETA, &_tty);
208#endif
209}

211/* md_heed_signals():
212 *
213 * This routine tells the program to call particular routines when
214 * certain interrupts/events occur:
215 *
216 * SIGINT: call onintr() to interrupt fight with monster or long rest.
217 * SIGQUIT: call byebye() to check for game termination.
218 * SIGHUP: call error_save() to save game when terminal hangs up.
219 *
220 * On VMS, SIGINT and SIGQUIT correspond to ^C and ^Y.
221 *
222 * This routine is not strictly necessary and can be stubbed. This will
223 * mean that the game cannot be interrupted properly with keyboard
224 * input, this is not usually critical.
225 */
226
227md_heed_signals()
228{
229 signal(SIGINT, onintr);
230 signal(SIGQUIT, byebye);
231 signal(SIGHUP, error_save);
232}
233
234/* md_ignore_signals():
235 *
236 * This routine tells the program to completely ignore the events mentioned
237 * in md_heed_signals() above. The event handlers will later be turned on
238 * by a future call to md_heed_signals(), so md_heed_signals() and
239 * md_ignore_signals() need to work together.
240 *
241 * This function should be implemented or the user risks interrupting
242 * critical sections of code, which could cause score file, or saved-game
243 * file, corruption.
244 */
245
246md_ignore_signals()
247{
248 signal(SIGQUIT, SIG_IGN);
249 signal(SIGINT, SIG_IGN);
250 signal(SIGHUP, SIG_IGN);
251}

292/* md_gct(): (Get Current Time)
293 *
294 * This function returns the current year, month(1-12), day(1-31), hour(0-23),
295 * minute(0-59), and second(0-59). This is used for identifying the time
296 * at which a game is saved.
297 *
298 * This function is not strictly necessary. It can be stubbed by returning
299 * zeros instead of the correct year, month, etc. If your operating
300 * system doesn't provide all of the time units requested here, then you
301 * can provide only those that it does, and return zeros for the others.
302 * If you cannot provide good time values, then users may be able to copy
303 * saved-game files and play them.
304 */
305
306md_gct(rt_buf)
307struct rogue_time *rt_buf;
308{
309 struct tm *t, *localtime();
310 long seconds;
311
312 time(&seconds);
313 t = localtime(&seconds);
314
315 rt_buf->year = t->tm_year;
316 rt_buf->month = t->tm_mon + 1;
317 rt_buf->day = t->tm_mday;
318 rt_buf->hour = t->tm_hour;
319 rt_buf->minute = t->tm_min;
320 rt_buf->second = t->tm_sec;
321}

323/* md_gfmt: (Get File Modification Time)
324 *
325 * This routine returns a file's date of last modification in the same format
326 * as md_gct() above.
327 *
328 * This function is not strictly necessary. It is used to see if saved-game
329 * files have been modified since they were saved. If you have stubbed the
330 * routine md_gct() above by returning constant values, then you may do
331 * exactly the same here.
332 * Or if md_gct() is implemented correctly, but your system does not provide
333 * file modification dates, you may return some date far in the past so
334 * that the program will never know that a saved-game file being modified.
335 * You may also do this if you wish to be able to restore games from
336 * saved-games that have been modified.
337 */
338
339md_gfmt(fname, rt_buf)
340char *fname;
341struct rogue_time *rt_buf;
342{
343 struct stat sbuf;
344 long seconds;
345 struct tm *t;
346
347 stat(fname, &sbuf);
348 seconds = (long) sbuf.st_mtime;
349 t = localtime(&seconds);
350
351 rt_buf->year = t->tm_year;
352 rt_buf->month = t->tm_mon + 1;
353 rt_buf->day = t->tm_mday;
354 rt_buf->hour = t->tm_hour;
355 rt_buf->minute = t->tm_min;
356 rt_buf->second = t->tm_sec;
357}

360 *
361 * This function deletes the specified file, and returns true (1) if the
362 * operation was successful. This is used to delete saved-game files
363 * after restoring games from them.
364 *
365 * Again, this function is not strictly necessary, and can be stubbed

372char *fname;

415/* md_getenv()
416 *
417 * This routine gets certain values from the user's environment. These
418 * values are strings, and each string is identified by a name. The names
419 * of the values needed, and their use, is as follows:
420 *
421 * TERMCAP
422 * The name of the users's termcap file, NOT the termcap entries
423 * themselves. This is used ONLY if the program is compiled with
424 * CURSES defined (-DCURSES). Even in this case, the program need
425 * not find a string for TERMCAP. If it does not, it will use the
426 * default termcap file as returned by md_gdtcf();
427 * TERM
428 * The name of the users's terminal. This is used ONLY if the program
429 * is compiled with CURSES defined (-DCURSES). In this case, the string
430 * value for TERM must be found, or the routines in curses.c cannot
431 * function, and the program will quit.
432 * ROGUEOPTS
433 * A string containing the various game options. This need not be
434 * defined.
435 * HOME
436 * The user's home directory. This is only used when the user specifies
437 * '~' as the first character of a saved-game file. This string need
438 * not be defined.
439 * SHELL
440 * The user's favorite shell. If not found, "/bin/sh" is assumed.
441 *
442 * If your system does not provide a means of searching for these values,
443 * you will have to do it yourself. None of the values above really need
444 * to be defined except TERM when the program is compiled with CURSES
445 * defined. In this case, as a bare minimum, you can check the 'name'
446 * parameter, and if it is "TERM" find the terminal name and return that,
447 * else return zero. If the program is not compiled with CURSES, you can
448 * get by with simply always returning zero. Returning zero indicates
449 * that their is no defined value for the given string.
450 */
451
452char *
453md_getenv(name)
454char *name;
455{
456 char *value;
457 char *getenv();
458
459 value = getenv(name);
460
461 return (value);
462}

479/* md_gseed() (Get Seed)
480 *
481 * This function returns a seed for the random number generator (RNG). This
482 * seed causes the RNG to begin generating numbers at some point in it's
483 * sequence. Without a random seed, the RNG will generate the same set
484 * of numbers, and every game will start out exactly the same way. A good
485 * number to use is the process id, given by getpid() on most UNIX systems.
486 *
487 * You need to find some single random integer, such as:
488 * process id.
489 * current time (minutes + seconds) returned from md_gct(), if implemented.
490 *
491 * It will not help to return "get_rand()" or "rand()" or the return value of
492 * any pseudo-RNG. If you don't have a random number, you can just return 1,
493 * but this means your games will ALWAYS start the same way, and will play
494 * exactly the same way given the same input.
495 */
496
497md_gseed()
498{
499 return (getpid());
500}

515/* md_lock():
516 *
517 * This function is intended to give the user exclusive access to the score
518 * file. It does so by flock'ing the score file. The full path name of the
519 * score file should be defined for any particular site in rogue.h. The
520 * constants _PATH_SCOREFILE defines this file name.
521 *
522 * When the parameter 'l' is non-zero (true), a lock is requested. Otherwise
523 * the lock is released.
524 */
525
526md_lock(l)
527boolean l;
528{
529 static int fd;
530 short tries;
531
532 if (l) {
533 if ((fd = open(_PATH_SCOREFILE, O_RDONLY)) < 1) {
534 message("cannot lock score file", 0);
535 return;
536 }
537 for (tries = 0; tries < 5; tries++)
538 if (!flock(fd, LOCK_EX | LOCK_NB)) {
539 return;
540 }
541 } else {
542 (void) flock(fd, LOCK_NB);
543 (void) close(fd);
544 }
545}

547/* md_shell():
548 *
549 * This function spawns a shell for the user to use. When this shell is
550 * terminated, the game continues. Since this program may often be run
551 * setuid to gain access to privileged files, care is taken that the shell
552 * is run with the user's REAL user id, and not the effective user id.
553 * The effective user id is restored after the shell completes.
554 */
555
556md_shell(shell)
557char *shell;
558{
559 long w[2];
560
561 if (!fork()) {
562 int uid;
563
564 uid = getuid();
565 setuid(uid);
566 execl(shell, shell, 0, NULL);
567 }
568 wait(w);
569}

577/* md_cbreak_no_echo_nonl:
578 *
579 * This routine sets up some terminal characteristics. The tty-driver
580 * must be told to:
581 * 1.) Not echo input.
582 * 2.) Transmit input characters immediately upon typing. (cbreak mode)
583 * 3.) Move the cursor down one line, without changing column, and
584 * without generating a carriage-return, when it
585 * sees a line-feed. This is only necessary if line-feed is ever
586 * used in the termcap 'do' (cursor down) entry, in which case,
587 * your system should must have a way of accomplishing this.
588 *
589 * When the parameter 'on' is true, the terminal is set up as specified
590 * above. When this parameter is false, the terminal is restored to the
591 * original state.
592 *
593 * Raw mode should not to be used. Keyboard signals/events/interrupts should
594 * be sent, although they are not strictly necessary. See notes in
595 * md_heed_signals().
596 *
597 * This function must be implemented for rogue to run properly if the
598 * program is compiled with CURSES defined to use the enclosed curses
599 * emulation package. If you are not using this, then this routine is
600 * totally unnecessary.
601 *
602 * Notice that information is saved between calls. This is used to
603 * restore the terminal to an initial saved state.
604 *
605 */
606
607md_cbreak_no_echo_nonl(on)
608boolean on;
609{
610#ifdef UNIX_BSD4_2
611 static struct sgttyb tty_buf;
612 static int tsave_flags;
613
614 if (on) {
615 ioctl(0, TIOCGETP, &tty_buf);
616 tsave_flags = tty_buf.sg_flags;
617 tty_buf.sg_flags |= CBREAK;
618 tty_buf.sg_flags &= ~(ECHO | CRMOD); /* CRMOD: see note 3 above */
619 ioctl(0, TIOCSETP, &tty_buf);
620 } else {
621 tty_buf.sg_flags = tsave_flags;
622 ioctl(0, TIOCSETP, &tty_buf);
623 }
624#endif
625#ifdef UNIX_SYSV
626 struct termio tty_buf;
627 static struct termio tty_save;
628
629 if (on) {
630 ioctl(0, TCGETA, &tty_buf);
631 tty_save = tty_buf;
632 tty_buf.c_lflag &= ~(ICANON | ECHO);
633 tty_buf.c_oflag &= ~ONLCR;
634 tty_buf.c_cc[4] = 1; /* MIN */
635 tty_buf.c_cc[5] = 2; /* TIME */
636 ioctl(0, TCSETAF, &tty_buf);
637 } else {
638 ioctl(0, TCSETAF, &tty_save);
639 }
640#endif
641}

643/* md_gdtcf(): (Get Default Termcap File)
644 *
645 * This function is called ONLY when the program is compiled with CURSES
646 * defined. If you use your system's curses/termlib library, this function
647 * won't be called. On most UNIX systems, "/etc/termcap" suffices.
648 *
649 * If their is no such termcap file, then return 0, but in that case, you
650 * must have a TERMCAP file returned from md_getenv("TERMCAP"). The latter
651 * will override the value returned from md_gdtcf(). If the program is
652 * compiled with CURSES defined, and md_gdtcf() returns 0, and
653 * md_getenv("TERMCAP") returns 0, the program will have no terminal
654 * capability information and will quit.
655 */
656
657char *
658md_gdtcf()
659{
660 return ("/etc/termcap");
661}
